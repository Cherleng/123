## 数据结构 - 循环单链表实现/插入/查找/删除

#include <stdio.h>
#include<stdlib.h>
#include<string.h>

typedef struct Node{
	int data;
	struct Node *next;
}node;

node *head,*ptr;

void init(){
	
	ptr = (node *)malloc(sizeof(node));
	ptr->data = 1;
	head = ptr;
	ptr = (node *)malloc(sizeof(node));
	ptr->data = 2;
	head->next = ptr;
	ptr = (node *)malloc(sizeof(node));
	ptr->data = 3;
	ptr->next=head;
	head->next->next = ptr;
	
	
}

void print_link(node *temp){
	if(!head){
		printf("链表不存在\n"); 
	}
	else{
	
		int count; 
		count = 1;
		while(temp->next!=head){
			printf("The %d-th data is %d\n",count,temp->data);
			temp = temp->next;
			count++;
		}
		printf("The %d-th data is %d\n",count,temp->data);
}
}
void find_num(){
	if(!head){
		printf("链表不存在！\n");
	}
	else{
	
		int data;
		int count = 1;
		int flag = 0;
		printf("Enter the nums\n");
		scanf("%d",&data);
		ptr = head;
		while(ptr->next!=head){
			if(ptr->data==data){
				printf("the data %d in %d-th\n",ptr->data,count);
				flag = 1;
				return;
				
			}
			ptr=ptr->next;
			count++;
		}
		if(ptr->data==data){
			printf("the data %d in %d-th\n",ptr->data,count);
			flag=1;
		}
		if(!flag){
			printf("链表中没有该数字！\n");
			
		}
		
	}
}

void insert_link_top(){
	int data;
	printf("Enter a data\n");
	scanf("%d",&data);
	ptr = (node *)malloc(sizeof(node));
	ptr->data = data;
	if(head){
		node *temp;
		temp = head;
		while(temp->next!=head){
			temp = temp->next;
		}
		ptr->next = head;
		temp->next=ptr;
		head = ptr;
}
	else{
		head = ptr;
		ptr->next = head;
	}
}

void insert_link_middle(){
	int data;
	int pos;
	if(head){
	
		printf("Enter the data\n");
		scanf("%d",&data);
		printf("Enter the Position\n");
		scanf("%d",&pos);
		node *temp;
		temp = head;
		int i;
		for(i=0;i<pos-1;i++){
			temp = temp->next;
		}
		ptr = (node *)malloc(sizeof(node));
		ptr->data = data;
		ptr->next =temp->next;
		temp->next=ptr;
		
}
	else{
		printf("该链表不存在，无法操作！\n");
	}
	}

void insert_link_end(){
	int data;
	printf("Enter a data\n");
	scanf("%d",&data);
	node *temp;
	ptr = (node *)malloc(sizeof(node));
	if(head){
	
		temp = head;
		while(temp->next!=head){
			temp = temp->next;
		}
		ptr->data = data;
		ptr->next = head;
		temp->next=ptr;
}
	else{
		printf("该链表为空 无法操作!\n");
	}
}

void delete_num_top(){
	node *temp;
	temp = head;
	if(head){
		while(temp->next!=head){
			temp = temp->next;
		}
		temp->next = head->next;
		ptr = head;
		head = ptr->next;
		free(ptr);
	}
	else{
		printf("该链表为空，无法操作！\n");
	}
	
}

void delete_num_end(){
	node *temp;
	node *p;
	temp = head;

		while(temp->next!=head){
			p = temp;
			temp = temp->next;

		p->next =head;

	}

}

void delete_num_middle(){
	int pos;
	printf("Enter the position\n");
	scanf("%d",&pos);
	if(head){
		node *temp;
		temp = head;
		int i;
		for(i=0;i<pos-1;i++){
			temp = temp->next;
		}
